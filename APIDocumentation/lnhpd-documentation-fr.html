Introduction

La Base de données des produits de santé naturels homologués (BDPSNH) contient des renseignements sur les produits de santé naturels ayant reçu une licence de mise en marché de Santé Canada.

Cette interface de programmation d'applications (API) permet aux concepteurs d'accéder à ces renseignements en format JSON, XML et CSV afin de les utiliser dans leurs propres applications.

L'adresse URI de base sur la Base de donnée de produits de santé naturels homologués (BDPSNH) est  https://api.hres.ca/lnhpd/api/. Toute requête est effectuée par l'entremise de cette adresse URI.

Exemple de jQuery

Exemple d'une fonctionnalité qui recherche les ingrédients selon le numéro d'identification d'une demande et la langue sélectionnée :

function getIngredient(id, lang) {
	var base = 'https://imsd.hres.ca';
	var uri = base + '/lnhpd/api/ingredient/?lang=' + lang + '&id=' + id;
	$.ajax({
		url: uri,
		type: 'GET',
		Accept: "application/json",
		dataType: 'json',
		success: function(data) {
			console.log(data.ingredient_name);
			var frag = document.createDocumentFragment();
			var h2 = document.createElement("h2");
			var IngredientName = document.createTextNode(data.ingredient_name);
			var p = document.createElement("p");
			var text = document.createTextNode(data.panels[0].text);
			h2.appendChild(IngredientName);
			p.appendChild(text);
			frag.appendChild(h2);
			frag.appendChild(p);
			$("#responses")[0].appendChild(frag);
		},
		error: function(error) {},
	});
	return;
};

function getProductLicence(id, lang) {
	var base = 'https://imsd.hres.ca';
	var uri = base + '/lnhpd/api/productlicence/?lang=' + lang + '&id=' + id;
	$.ajax({
		url: uri,
		type: 'GET',
		Accept: "application/json",
		dataType: 'json',
		success: function(data) {
			console.log(data.product_licence_name);
			var frag = document.createDocumentFragment();
			var h2 = document.createElement("h2");
			var ProductLicenceName = document.createTextNode(data.product_licence_name);
			var p = document.createElement("p");
			var text = document.createTextNode(data.panels[0].text);
			h2.appendChild(ProductLicenceName);
			p.appendChild(text);
			frag.appendChild(h2);
			frag.appendChild(p);
			$("#responses")[0].appendChild(frag);
		},
		error: function(error) {},
	});
	return;
};
