Introduction

The Drug and Health Products Register contains information about .....

This Application Programming Interface (API) allows developers to access that information in JSON, XML and CSV format for reuse in their own applications.

The base URI for the Drug and Health Products Register  https://api.hres.ca/dhpr1/api/. Any requests are made relative to this URI.

jQuery example

Example of a function that search for an active ingredient given an identification number and language selection:

function getActiveIngredient(id, lang) {
      var base = 'https://imsd.hres.ca';
      var uri = base + '/dhpr1/api/activeingredient/?lang=' + lang + '&id=' + id;
      $.ajax({
              url:uri,
              type:'GET',
              Accept:"application/json",
              dataType: 'json',
              success:function(data){
                           console.log(data.ingredient);
                           var frag = document.createDocumentFragment();
                           var h2 = document.createElement("h2");
                           var ActiveIngredientName = document.createTextNode(data.ingredient);
                           var p = document.createElement("p");
                           var text = document.createTextNode(data.panels[0].text);
                           h2.appendChild(ActiveIngredientName);
                           p.appendChild(text);
                           frag.appendChild(h2);
                           frag.appendChild(p);
                           $("#responses")[0].appendChild(frag);
              },
              error:function(error){
              },
      });
      return;
 };
