Introduction

The Drug and Health Products Register contains information about .....

Cette interface de programmation d'applications (API) permet aux concepteurs d'accéder à ces renseignements en format JSON, XML et CSV afin de les utiliser dans leurs propres applications.

L'adresse URI de base du Registre des médicaments et des produits de santé est  https://api.hres.ca/dhpr1/api/. Toute requête est effectuée par l'entremise de cette adresse URI.

Exemple de jQuery

Exemple d'une fonctionnalité qui recherche un ACTIVE INGREDIENT selon un numéro d'identification et la langue sélectionnée :

function getActiveIngredient(id, lang) {
      var base = 'https://imsd.hres.ca';
      var uri = base + '/dhpr1/api/activeingredient/?lang=' + lang + '&id=' + id;
      $.ajax({
              url:uri,
              type:'GET',
              Accept:"application/json",
              dataType: 'json',
              success:function(data){
                           console.log(data.ingredient);
                           var frag = document.createDocumentFragment();
                           var h2 = document.createElement("h2");
                           var ActiveIngredientName = document.createTextNode(data.ingredient);
                           var p = document.createElement("p");
                           var text = document.createTextNode(data.panels[0].text);
                           h2.appendChild(ActiveIngredientName);
                           p.appendChild(text);
                           frag.appendChild(h2);
                           frag.appendChild(p);
                           $("#responses")[0].appendChild(frag);
             },
             error:function(error){
             },
      });
     return;
 };
